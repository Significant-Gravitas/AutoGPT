13def build_default_prompt_generator() -> PromptGenerator: 

14 """ 

15 This function generates a prompt string that includes various constraints, 

16 commands, resources, and performance evaluations. 

17 

18 Returns: 

19 str: The generated prompt string. 

20 """ 

21 

22 # Initialize the PromptGenerator object 

23 prompt_generator = PromptGenerator() 

24 

25 # Add constraints to the PromptGenerator object 

26 prompt_generator.add_constraint( 

27 "~4000 word limit for short term memory. Your short term memory is short, so" 

28 " immediately save important information to files." 

29 ) 

30 prompt_generator.add_constraint( 

31 "If you are unsure how you previously did something or want to recall past" 

32 " events, thinking about similar events will help you remember." 

33 ) 

34 prompt_generator.add_constraint("No user assistance") 

35 prompt_generator.add_constraint( 

36 'Exclusively use the commands listed in double quotes e.g. "command name"' 

37 ) 

38 

39 # Define the command list 

40 commands = [ 

41 ("Do Nothing", "do_nothing", {}), 

42 ("Task Complete (Shutdown)", "task_complete", {"reason": "<reason>"}), 

43 ] 

44 

45 # Add commands to the PromptGenerator object 

46 for command_label, command_name, args in commands: 

47 prompt_generator.add_command(command_label, command_name, args) 

48 

49 # Add resources to the PromptGenerator object 

50 prompt_generator.add_resource( 

51 "Internet access for searches and information gathering." 

52 ) 

53 prompt_generator.add_resource("Long Term memory management.") 

54 prompt_generator.add_resource( 

55 "GPT-3.5 powered Agents for delegation of simple tasks." 

56 ) 

57 prompt_generator.add_resource("File output.") 

58 

59 # Add performance evaluations to the PromptGenerator object 

60 prompt_generator.add_performance_evaluation( 

61 "Continuously review and analyze your actions to ensure you are performing to" 

62 " the best of your abilities." 

63 ) 

64 prompt_generator.add_performance_evaluation( 

65 "Constructively self-criticize your big-picture behavior constantly." 

66 ) 

67 prompt_generator.add_performance_evaluation( 

68 "Reflect on past decisions and strategies to refine your approach." 

69 ) 

70 prompt_generator.add_performance_evaluation( 

71 "Every command has a cost, so be smart and efficient. Aim to complete tasks in" 

72 " the least number of steps." 

73 ) 

74 prompt_generator.add_performance_evaluation("Write all code to a file.") 

75 return prompt_generator
