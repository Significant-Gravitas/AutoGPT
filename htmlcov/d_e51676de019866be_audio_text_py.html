20def read_audio_from_file(filename: str) -> str: 

21 """ 

22 Convert audio to text. 

23 

24 Args: 

25 audio_path (str): The path to the audio file 

26 

27 Returns: 

28 str: The text from the audio

36def read_audio(audio: bytes) -> str: 

37 """ 

38 Convert audio to text. 

39 

40 Args: 

41 audio (bytes): The audio to convert 

42 

43 Returns: 

44 str: The text from the audio 

45 """ 

46 model = CFG.huggingface_audio_to_text_model 

47 api_url = f"https://api-inference.huggingface.co/models/{model}" 

48 api_token = CFG.huggingface_api_token 

49 headers = {"Authorization": f"Bearer {api_token}"} 

50 

51 if api_token is None: 

52 raise ValueError( 

53 "You need to set your Hugging Face API token in the config file." 

54 ) 

55 

56 response = requests.post( 

57 api_url, 

58 headers=headers, 

59 data=audio, 

60 ) 

61 

62 text = json.loads(response.content.decode("utf-8"))["text"] 

63 return f"The audio says: {text}"
