15@command("execute_python_file", "Execute Python File", '"filename": "<filename>"') 

16def execute_python_file(filename: str) -> str: 

17 """Execute a Python file in a Docker container and return the output 

18 

19 Args: 

20 filename (str): The name of the file to execute 

21 

22 Returns: 

23 str: The output of the file 

24 """ 

25 file = filename 

26 print(f"Executing file '{file}' in workspace '{WORKSPACE_PATH}'") 

27 

28 if not file.endswith(".py"): 

29 return "Error: Invalid file type. Only .py files are allowed." 

30 

31 file_path = path_in_workspace(file) 

32 

33 if not os.path.isfile(file_path): 

34 return f"Error: File '{file}' does not exist." 

35 

36 if we_are_running_in_a_docker_container(): 

37 result = subprocess.run( 

38 f"python {file_path}", capture_output=True, encoding="utf8", shell=True 

39 ) 

40 if result.returncode == 0: 

41 return result.stdout 

42 else: 

43 return f"Error: {result.stderr}" 

44 

45 try: 

46 client = docker.from_env() 

47 

48 # You can replace this with the desired Python image/version 

49 # You can find available Python images on Docker Hub: 

50 # https://hub.docker.com/_/python 

51 image_name = "python:3-alpine" 

52 try: 

53 client.images.get(image_name) 

54 print(f"Image '{image_name}' found locally") 

55 except ImageNotFound: 

56 print(f"Image '{image_name}' not found locally, pulling from Docker Hub") 

57 # Use the low-level API to stream the pull response 

58 low_level_client = docker.APIClient() 

59 for line in low_level_client.pull(image_name, stream=True, decode=True): 

60 # Print the status and progress, if available 

61 status = line.get("status") 

62 progress = line.get("progress") 

63 if status and progress: 

64 print(f"{status}: {progress}") 

65 elif status: 

66 print(status) 

67 

68 container = client.containers.run( 

69 image_name, 

70 f"python {file}", 

71 volumes={ 

72 os.path.abspath(WORKSPACE_PATH): { 

73 "bind": "/workspace", 

74 "mode": "ro", 

75 } 

76 }, 

77 working_dir="/workspace", 

78 stderr=True, 

79 stdout=True, 

80 detach=True, 

81 ) 

82 

83 container.wait() 

84 logs = container.logs().decode("utf-8") 

85 container.remove() 

86 

87 # print(f"Execution complete. Output: {output}") 

88 # print(f"Logs: {logs}") 

89 

90 return logs 

91 

92 except docker.errors.DockerException as e: 

93 print( 

94 "Could not run the script in a container. If you haven't already, please install Docker https://docs.docker.com/get-docker/" 

95 ) 

96 return f"Error: {str(e)}" 

97 

98 except Exception as e: 

99 return f"Error: {str(e)}"

113

116 

117 Returns: 

118 str: The output of the command 

119 """ 

120 

121 if not CFG.execute_local_commands:

123 "You are not allowed to run local shell commands. To execute"

125 "in your config. Do not attempt to bypass the restriction." 

126 ) 

127 current_dir = os.getcwd() 

128 # Change dir into workspace if necessary 

129 if str(WORKSPACE_PATH) not in current_dir: 

130 os.chdir(WORKSPACE_PATH) 

131 

132 print(f"Executing command '{command_line}' in working directory '{os.getcwd()}'") 

133 

134 result = subprocess.run(command_line, capture_output=True, shell=True) 

135 output = f"STDOUT:\n{result.stdout}\nSTDERR:\n{result.stderr}" 

136 

137 # Change back to whatever the prior working dir was 

138 

139 os.chdir(current_dir)

151def execute_shell_popen(command_line) -> str: 

152 """Execute a shell command with Popen and returns an english description 

153 of the event and the process id 

154 

155 Args: 

156 command_line (str): The command line to execute 

157 

158 Returns: 

159 str: Description of the fact that the process started and its id 

160 """ 

161 current_dir = os.getcwd() 

162 # Change dir into workspace if necessary 

163 if str(WORKSPACE_PATH) not in current_dir: 

164 os.chdir(WORKSPACE_PATH) 

165 

166 print(f"Executing command '{command_line}' in working directory '{os.getcwd()}'") 

167 

168 do_not_show_output = subprocess.DEVNULL 

169 process = subprocess.Popen( 

170 command_line, shell=True, stdout=do_not_show_output, stderr=do_not_show_output 

171 ) 

172 

173 # Change back to whatever the prior working dir was 

174 

175 os.chdir(current_dir) 

176 

177 return f"Subprocess started with PID:'{str(process.pid)}'"
