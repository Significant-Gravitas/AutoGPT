1"""Git operations for autogpt""" 

2from git.repo import Repo 

3 

4from autogpt.commands.command import command 

5from autogpt.config import Config 

6from autogpt.workspace import path_in_workspace 

7 

8CFG = Config() 

9 

10 

11@command( 

12 "clone_repository", 

13 "Clone Repository", 

14 '"repository_url": "<repository_url>", "clone_path": "<clone_path>"', 

15 CFG.github_username and CFG.github_api_key, 

16 "Configure github_username and github_api_key.", 

17) 

18def clone_repository(repository_url: str, clone_path: str) -> str: 

19 """Clone a GitHub repository locally. 

20 

21 Args: 

22 repository_url (str): The URL of the repository to clone. 

23 clone_path (str): The path to clone the repository to. 

24 

25 Returns: 

26 str: The result of the clone operation. 

27 """ 

28 split_url = repository_url.split("//") 

29 auth_repo_url = f"//{CFG.github_username}:{CFG.github_api_key}@".join(split_url) 

30 safe_clone_path = path_in_workspace(clone_path) 

31 try: 

32 Repo.clone_from(auth_repo_url, safe_clone_path) 

33 return f"""Cloned {repository_url} to {safe_clone_path}""" 

34 except Exception as e: 

35 return f"Error: {str(e)}"
