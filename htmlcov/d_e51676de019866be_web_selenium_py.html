29@command( 

30 "browse_website", 

31 "Browse Website", 

32 '"url": "<url>", "question": "<what_you_want_to_find_on_website>"', 

33) 

34def browse_website(url: str, question: str) -> tuple[str, WebDriver]: 

35 """Browse a website and return the answer and links to the user 

36 

37 Args: 

38 url (str): The url of the website to browse 

39 question (str): The question asked by the user 

40 

41 Returns: 

42 Tuple[str, WebDriver]: The answer and links to the user and the webdriver 

43 """ 

44 driver, text = scrape_text_with_selenium(url) 

45 add_header(driver) 

46 summary_text = summary.summarize_text(url, text, question, driver) 

47 links = scrape_links_with_selenium(driver, url) 

48 

49 # Limit links to 5 

50 if len(links) > 5: 

51 links = links[:5] 

52 close_browser(driver) 

53 return f"Answer gathered from website: {summary_text} \n \n Links: {links}", driver

56def scrape_text_with_selenium(url: str) -> tuple[WebDriver, str]: 

57 """Scrape text from a website using selenium 

58 

59 Args: 

60 url (str): The url of the website to scrape 

61 

62 Returns: 

63 Tuple[WebDriver, str]: The webdriver and the text scraped from the website 

64 """ 

65 logging.getLogger("selenium").setLevel(logging.CRITICAL) 

66 

67 options_available = { 

68 "chrome": ChromeOptions, 

69 "safari": SafariOptions, 

70 "firefox": FirefoxOptions, 

71 } 

72 

73 options = options_available[CFG.selenium_web_browser]() 

74 options.add_argument( 

75 "user-agent=Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/112.0.5615.49 Safari/537.36" 

76 ) 

77 

78 if CFG.selenium_web_browser == "firefox": 

79 driver = webdriver.Firefox( 

80 executable_path=GeckoDriverManager().install(), options=options 

81 ) 

82 elif CFG.selenium_web_browser == "safari": 

83 # Requires a bit more setup on the users end 

84 # See https://developer.apple.com/documentation/webkit/testing_with_webdriver_in_safari 

85 driver = webdriver.Safari(options=options) 

86 else: 

87 if platform == "linux" or platform == "linux2": 

88 options.add_argument("--disable-dev-shm-usage") 

89 options.add_argument("--remote-debugging-port=9222") 

90 

91 options.add_argument("--no-sandbox") 

92 if CFG.selenium_headless: 

93 options.add_argument("--headless") 

94 options.add_argument("--disable-gpu") 

95 

96 driver = webdriver.Chrome( 

97 executable_path=ChromeDriverManager().install(), options=options 

98 ) 

99 driver.get(url) 

100 

101 WebDriverWait(driver, 10).until( 

102 EC.presence_of_element_located((By.TAG_NAME, "body")) 

103 ) 

104 

105 # Get the HTML content directly from the browser's DOM 

106 page_source = driver.execute_script("return document.body.outerHTML;") 

107 soup = BeautifulSoup(page_source, "html.parser") 

108 

109 for script in soup(["script", "style"]): 

110 script.extract() 

111 

112 text = soup.get_text() 

113 lines = (line.strip() for line in text.splitlines()) 

114 chunks = (phrase.strip() for line in lines for phrase in line.split(" ")) 

115 text = "\n".join(chunk for chunk in chunks if chunk) 

116 return driver, text

119def scrape_links_with_selenium(driver: WebDriver, url: str) -> list[str]: 

120 """Scrape links from a website using selenium 

121 

122 Args: 

123 driver (WebDriver): The webdriver to use to scrape the links 

124 

125 Returns: 

126 List[str]: The links scraped from the website 

127 """ 

128 page_source = driver.page_source 

129 soup = BeautifulSoup(page_source, "html.parser") 

130 

131 for script in soup(["script", "style"]): 

132 script.extract() 

133 

134 hyperlinks = extract_hyperlinks(soup, url) 

135 

136 return format_hyperlinks(hyperlinks)
